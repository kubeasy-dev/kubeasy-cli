wait.PollUntilContextTimeout(ctx, 5*time.Second, 60*time.Second, true, func(ctx context.Context) (bool, error) {
    ep, err := clientset.CoreV1().Endpoints("cert-manager").Get(ctx, "cert-manager-webhook", metav1.GetOptions{})
    if err != nil { return false, nil }
    for _, subset := range ep.Subsets {
        if len(subset.Addresses) > 0 { return true, nil }
    }
    return false, nil
})

    2. installCertManager(ctx context.Context, clientset kubernetes.Interface, dynamicClient dynamic.Interface, mapper meta.RESTMapper) ComponentResult:

    Flow:
    ```
    // Idempotency check
    ready, _ := isCertManagerReadyWithClient(ctx, clientset)
    if ready { return ComponentResult{Name: "cert-manager", Status: StatusReady, Message: "already installed"} }

    // Two-pass apply
    // Pass 1: CRDs
    crdsManifest, err := kube.FetchManifest(certManagerCRDsURL())
    if err != nil { return notReady("cert-manager", err) }
    kube.CreateNamespace(ctx, clientset, certManagerNamespace)
    if err := kube.ApplyManifest(ctx, crdsManifest, certManagerNamespace, mapper, dynamicClient); err != nil {
        return notReady("cert-manager", err)
    }

    // Pass 2: controller (cert-manager.yaml includes CRDs too — this is idempotent)
    ctrlManifest, err := kube.FetchManifest(certManagerInstallURL())
    if err != nil { return notReady("cert-manager", err) }
    if err := kube.ApplyManifest(ctx, ctrlManifest, certManagerNamespace, mapper, dynamicClient); err != nil {
        return notReady("cert-manager", err)
    }

    // Wait for deployments
    deployments := []string{"cert-manager", "cert-manager-cainjector", "cert-manager-webhook"}
    if err := kube.WaitForDeploymentsReady(ctx, clientset, certManagerNamespace, deployments); err != nil {
        return notReady("cert-manager", err)
    }

    // Extra webhook endpoint polling
    if err := waitForCertManagerWebhookEndpoints(ctx, clientset); err != nil {
        return notReady("cert-manager", err)
    }

    return ComponentResult{Name: "cert-manager", Status: StatusReady}
    ```
