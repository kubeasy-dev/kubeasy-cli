    ```
    Step 1 — Build tlsCfg for the HTTP transport:
        tlsCfg := &tls.Config{}
        if target.TLS != nil && target.TLS.InsecureSkipVerify {
            tlsCfg.InsecureSkipVerify = true //nolint:gosec // intentional: self-signed certs in Kind cluster
        }

    Step 2 — If explicit TLS checks requested (validateExpiry or validateSANs) AND NOT insecureSkipVerify:
        Parse host:port from target.URL via url.Parse
        Use tls.Dialer{Config: &tls.Config{InsecureSkipVerify: true}} //nolint:gosec
        DialContext(reqCtx, "tcp", net.JoinHostPort(host, port))
        On dial error → return false, fmt.Sprintf("TLS dial failed: %v", err)
        cert := conn.ConnectionState().PeerCertificates[0]
        conn.Close()

        If ValidateExpiry:
            if time.Now().After(cert.NotAfter):
                delta := time.Since(cert.NotAfter)
                days := int(delta.Hours() / 24)
                return false, fmt.Sprintf("Certificate expired on %s (%d days ago)", cert.NotAfter.Format("2006-01-02"), days)

        If ValidateSANs:
            hostname := url.Parse(target.URL).Hostname()
            if target.HostHeader != "":
                hostname = target.HostHeader
            if err := cert.VerifyHostname(hostname); err != nil:
                return false, fmt.Sprintf("Hostname %q not in SANs: %v", hostname, cert.DNSNames)

    Step 3 — Build HTTP client with TLS transport:
        (same as current client but add TLSClientConfig: tlsCfg to http.Transport)
        client := &http.Client{
            Timeout: ...,
            Transport: &http.Transport{TLSClientConfig: tlsCfg},
            CheckRedirect: ..., // preserve existing ErrUseLastResponse
        }

    Step 4 — Make HTTP request (existing logic, unchanged):
        req, err := http.NewRequestWithContext(...)
        if target.HostHeader != "": req.Host = target.HostHeader
        resp, err := client.Do(req)
        // existing blocked/status-code logic unchanged
    ```
